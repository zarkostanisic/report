Sub chartsTcom()
'
' chartsTcom Macro
'
'
Dim objShape As InlineShape

' Number formats
decimalPercentFormat = "0.00%"
kwFormat = "0 kW"


For Each objShape In ActiveDocument.InlineShapes
    With objShape.Chart
        HasLegend = .HasLegend

        ' 3D pie
        If .ChartType = xl3DPie Then
            Title = .ChartTitle.Text
            .ChartStyle = 3
            .ChartColor = 21

            .ApplyLayout (3)

            ' Title
            .HasTitle = True
            .ChartTitle.Text = Title

            ' Legend
            .Legend.Position = xlLegendPositionBottom
            .Legend.Left = 0
            .Legend.Width = 200

            .ChartArea.Format.ThreeD.RotationY = 30
        End If

        ' Column and column clustered
        If .ChartType = xlColumnClustered Then
            ' Column clustered
            .ChartStyle = 11
            ' Column
            If .SeriesCollection.Count = 1 Then
                .ChartStyle = 3
            End If

            .ApplyLayout (3)

            ' Horizontal(Category) Axis Major Gridlines
            With .Axes(xlCategory)
                .HasMajorGridlines = True
                .MajorGridlines.Border.Color = RGB(128, 128, 128)
            End With

            .PlotArea.Border.Color = RGB(128, 128, 128)

        End If

        ' Data label show value
        .ApplyDataLabels xlDataLabelsShowValue

        ' Set font and border color
        With .ChartArea
            .Font.Color = RGB(128, 128, 128)
            .Border.Color = RGB(128, 128, 128)
        End With

        ' Title
        If .HasTitle = True Then
            .ChartTitle.Font.Size = 10

            ' Hide title
            If .ChartTitle.Text = "#no-title" Then
                .ChartTitle.Text = " "
            End If
        End If

        ' Hide legend
        If HasLegend = False Then
            .HasLegend = False
        End If

    End With
Next
End Sub
